Cmodule::IncludeModule('iblock');

$arFilter = array('IBLOCK_ID' => 3);
# записываем элементы в массив
$res = CIBlockElement::GetList(false, $arFilter, array('ID'));
while($el = $res->GetNext())
{
	$arElementsID[] = $el['ID'];
}
# обновляем свойство элемента
foreach($arElementsID as $ELEMENT_ID)
{
	$PROPERTY_CODE = "EI";  # код свойства
	$PROPERTY_VALUE = "Карат";  # значение свойства
	CIBlockElement::SetPropertyValuesEx($ELEMENT_ID, false, array($PROPERTY_CODE => $PROPERTY_VALUE));
